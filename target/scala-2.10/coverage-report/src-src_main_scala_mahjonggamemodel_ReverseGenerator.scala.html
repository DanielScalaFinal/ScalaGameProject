10   def scramble(game:Game) {
11     var reversed = List[TilePair]()
12     val random = new Random()
13     while (game.tiles.size > 0) {
14       // Find two or more random outer tiles, remove them and store the coordinates
15       var removables = extractRemovableTiles(game)
16       // Continue until no more random removable tile pairs are left in the list
17       while (removables.length > 1) {
18         val index1 = random.nextInt(removables.length)
19         var index2 = random.nextInt(removables.length)
20         while (index1 == index2) index2 = (index2 + 1) % removables.length
21         val pair = new TilePair(removables(index1), removables(index2))
22         reversed = pair :: reversed
23         removables = removables.filter(t => t != pair.tile1 && t != pair.tile2)
24       }
25       // Add remaining tiles to the field
26       removables.foreach(game += _)
27     }
28     // Read the list from behind and and get random a random tile type for each pair to build the game
29     var typeRand = new Random().nextInt(game.tileTypes.length)
30     for (pair <- reversed) {
31       val typeIndex = typeRand % game.tileTypes.length
32       typeRand += 1
33       pair.tile1.tileType = game.tileTypes(typeIndex)
34       pair.tile2.tileType = game.tileTypes(typeIndex)
35       game += pair.tile1
36       game += pair.tile2
37     }
38   }
